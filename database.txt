# Bronx Science Olympiad Database Schema

## Collection: Members
- id: string (auto-generated by Firebase)
- firstName: string
- lastName: string
- email: string (unique identifier)
- grade: string (9-12)
- password: string (min 8 chars, store securely in production)
- memberStatus: string (enum: "new", "returning")
- events: string[] (array of event names)
- createdAt: timestamp
- status: string (enum: "pending", "active", "inactive")
- interestReason: string (for new members)
- pastExperience: string (for returning members)
- returnReason: string (for returning members)
- profilePicUrl: string (Firebase Storage URL to profile image)
- bio: string (user biography/about me)
- displayName: string (optional username for display)
- notificationPreferences: {
    email: boolean,
    reminders: boolean
  }
- lastLogin: timestamp
- competitionResults: [
    {
      competitionId: string,
      competitionName: string,
      date: timestamp,
      type: string (enum: "invitational", "regional", "state", "national"),
      eventResults: [
        {
          eventName: string,
          placement: number,
          outOf: number,
          medal: boolean,
          medalType: string (enum: "gold", "silver", "bronze", null)
        }
      ],
      teamRank: number,
      teamOutOf: number
    }
  ]

## Collection: Binders
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- eventName: string
- title: string
- description: string
- createdAt: timestamp
- updatedAt: timestamp
- contentUrl: string (Firebase Storage reference to text content)
- sections: array {
    title: string,
    order: number
  }

## Collection: Competitions
- id: string (auto-generated by Firebase)
- name: string
- date: timestamp
- type: string (enum: "invitational", "regional", "state", "national")
- location: string
- description: string
- registrationDeadline: timestamp
- events: string[] (array of available events)
- teamId: string (reference to team participating)
- results: {
    teamRank: number,
    teamOutOf: number,
    eventResults: [
      {
        eventName: string,
        participants: string[] (references to Member ids),
        placement: number,
        outOf: number
      }
    ]
  }

## Collection: Duosmium
- id: string (auto-generated by Firebase)
- url: string (original Duosmium results URL)
- competitionName: string
- date: timestamp
- location: string
- teams: [
    {
      teamName: string (e.g., "Bronx H.S. of Science A NY")
      rank: number
      totalScore: number
      events: [
        {
          eventName: string
          place: number
          participants: string[] (Member IDs who participated)
        }
      ]
    }
  ]
- parsedAt: timestamp 

## Collection: Tests
- id: string (auto-generated by Firebase)
- eventName: string
- testName: string
- description: string
- year: number
- competitionName: string (optional)
- competitionType: string (enum: "invitational", "regional", "state", "national", "practice")
- uploadedBy: string (reference to Member id)
- fileUrl: string (Firebase Storage reference to PDF file)
- fileSize: number (in bytes)
- uploadedAt: timestamp
- viewCount: number
- tags: string[] (for categorization/search)
- visibility: string (enum: "public", "team", "private")
