# Bronx Science Olympiad Database Schema

## Collection: Members
- id: string (auto-generated by Firebase)
- firstName: string
- lastName: string
- doeEmail: string (unique identifier)
- personalEmail: string
- phoneNumber: number
- grade: string (9-12)
- password: string (min 8 chars, store securely in production)

- memberStatus: string (enum: "new", "returning")
- createdAt: timestamp
- bio: string ("Tell us about yourself", all members)
- interestReason: string (for new members)
- howDidYouHearAboutUs: string (for new members)
- pastExperience: string (for returning members)
- returnReason: string (for returning members)
- yearsInTeam: number (for returning members)

- preseasonHrs: number (for returning members)
- regseasonHrs: number (for returning members)
- postseasonHrs: number (for returning members)

- profilePicUrl: string (Firebase Storage URL to profile image)
- importantNotification: string (enum: "none", "email", "text", "both")
- teamNotification: string (enum: "none", "email", "text", "both")
- eventNotification: string (enum: "none", "email", "text", "both")

- adminStatus: string (enum: "none", "full")


## Collection: Binders
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- eventName: string
- title: string
- description: string
- createdAt: timestamp
- updatedAt: timestamp
- contentUrl: string (Firebase Storage reference to text content)
- sections: array {
    title: string,
    order: number
  }

## Collection: Competitions
- id: string (auto-generated by Firebase)
- name: string
- date: timestamp
- type: string (enum: "invitational", "satellite", "official")
- location: string
- description: string
- registrationOpens: timestamp
- registrationDeadline: timestamp
- teamsRelease: timestamp
- numTeams: number
- teamPlacement: [{
  team: string (enum: "A", "B", "C"),
  event: string (name of event)
  participants: string[] (array of: member ids OR full name, if member is not in the database)
}] (optional, post-competition)
- results: {
    teamRank: number,
    teamOutOf: number,
    eventResults: [
      {
        eventName: string,
        team: string (enum: "A", "B", "C"),
        placement: number,
        outOf: number
      }
    ]
  } (optional, post-competition)
- duosmium: string (link to duosmium results) (optional, post-competition)


## Collection: Applications
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- competitionId: string (reference to Competitions)
- events: string[] (array of selected events)
- questions: {
    whyCompete: string,
    preparationPlan: string,
    firstTimePrep: string (optional),
    pastInfluence: string (optional)
  }
- status: string (enum: "pending", "approved", "rejected")
- teamPlacement: string (enum: "A", "B", "C")
- appliedAt: timestamp
- reviewedAt: timestamp (optional)
- reviewNotes: string (optional) 

## Collection: Meeting
- id: string (auto-generated by Firebase)
- date: timestamp
- eventName: string
- code: string (5 character unique code for the meeting)
- attended: string[] (array of member ids)


## Collection: LearningModules
- id: string (auto-generated by Firebase)
- eventName: string (full event name)
- title: string
- duration: string (e.g., "15 min", "45 min")
- unit: string (e.g., "Unit 1", "Unit 2")
- order: number (sequence in learning path)
- points: number (points for completion)
- prerequisites: string[] (array of module IDs that must be completed first)
- content: {
    overview: string,
    objectives: string[],
    resources: [
      {
        title: string,
        description: string,
        url: string,
        type: string
      }
    ]
  }
- validationType: string (enum: "none", "notesUpload", "problems", "AIconversation")
- problems: string[] (array of problems, optional)
- systemPrompt: string (system prompt for AI, optional)
- createdAt: timestamp

## Collection: Events
- id: string (auto-generated by Firebase)
- eventName: string (unique identifier for the event)
- description: string (general description)
- subject: string (enum: "biology", "chemistry", "physics", "earth-science", "construction-build", "precision-build", "classification-compilation")
- icon: string (FontAwesome icon class)
- generalInfo: {
    welcome: string,
    about: string,
    contentOverview: string,
    tips: string,
    resources: string[{
      title: string,
      description: string,
      url: string
    }]
  }
- meetingDay: string (enum: "Monday", "Tuesday", "Wednesday", "Thursday", "Friday")
- meetingBlock: number (enum: 1, 2)
- meetingRoom: number
- members: string[] (array of member ids)
- eventManagers: string[] (array of member ids)
- subjectDirectors: string[] (array of member ids)

## Collection: CalendarEvents
- id: string (auto-generated by Firebase)
- title: string
- description: string
- date: timestamp
- startTime: timestamp (optional)
- endTime: timestamp (optional)
- location: string (optional)
- eventType: string (enum: "meeting", "competition", "deadline", "workshop", "practice", "celebration", "no-school")



## Collection: ModuleCompletions
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- moduleId: string (reference to LearningModules)
- eventName: string
- totalPoints: number (points for completion)
- pointsGiven: number (points given by admin)
- completed: boolean
- completedAt: timestamp
- c3certified: boolean
- c2certified: boolean
- timeSpent: number (in minutes)
- c3method: string (enum: "none", "notesUpload", "problems", "AIconversation")
- notesFileUrl: string (Firebase Storage URL to notes file, optional)
- problemResponses: string[] (array of problem responses, optional)
- aiConversation: string[] (array of AI conversation messages)

## Collection: Portfolio
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- kind: string (enum: "artifact", "weekly", "diagnostic", "competition", "app_question")
- eventName: string (optional)
- createdAt: timestamp
- updatedAt: timestamp

- For kind "artifact":
  - artifactType: string (enum: "file", "link", "conversation")
  - title: string
  - url: string (for link/file)
  - storagePath: string (for file)
  - contentType: string (for file)
  - moduleIds: string[] (optional)
  - shareUrl: string (for conversation)
  - transcript: string (for conversation)

- For kind "weekly":
  - dateISO: string (YYYY-MM-DD)
  - content: string

- For kind "diagnostic":
  - diagnosticDateISO: string (YYYY-MM-DD)
  - reflection: string

- For kind "competition":
  - competitionName: string
  - whatWentWell: string
  - whatToImprove: string
  - reflection: string

- For kind "app_question":
  - questionId: string (e.g., q1..q5)
  - questionText: string
  - answer: string

## Collection: HouseCupThemes
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- userName: string (full name of submitter)
- theme: string (max 25 characters)
- houseNames: string[] (array of 4 house names, each max 20 characters)
- createdAt: timestamp
- votes: {
    first: number (count of 1st place votes),
    second: number (count of 2nd place votes),
    third: number (count of 3rd place votes)
  }

## Collection: HouseCupVotes
- id: string (userId - document ID matches user ID)
- userId: string (reference to Members)
- votes: {
    first: string (reference to HouseCupThemes ID or null),
    second: string (reference to HouseCupThemes ID or null),
    third: string (reference to HouseCupThemes ID or null)
  }
- updatedAt: timestamp
