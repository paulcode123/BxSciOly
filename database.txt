# Bronx Science Olympiad Database Schema

## Collection: Members
- id: string (auto-generated by Firebase)
- firstName: string
- lastName: string
- email: string (unique identifier)
- grade: string (9-12)
- password: string (min 8 chars, store securely in production)
- memberStatus: string (enum: "new", "returning")
- events: string[] (array of event names)
- createdAt: timestamp
- status: string (enum: "pending", "active", "inactive")
- interestReason: string (for new members)
- pastExperience: string (for returning members)
- returnReason: string (for returning members)
- profilePicUrl: string (Firebase Storage URL to profile image)
- bio: string (user biography/about me)
- displayName: string (optional username for display)
- notificationPreferences: {
    email: boolean,
    reminders: boolean
  }
- lastLogin: timestamp
- attendanceHistory: [
    {
      date: timestamp,
      eventId: string (reference to AttendanceEvent),
      eventName: string,
      checkInTime: timestamp
    }
  ]
- competitionResults: [
    {
      competitionId: string,
      competitionName: string,
      date: timestamp,
      type: string (enum: "invitational", "regional", "state", "national"),
      eventResults: [
        {
          eventName: string,
          placement: number,
          outOf: number,
          medal: boolean,
          medalType: string (enum: "gold", "silver", "bronze", null)
        }
      ],
      teamRank: number,
      teamOutOf: number
    }
  ]

## Collection: Binders
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- eventName: string
- title: string
- description: string
- createdAt: timestamp
- updatedAt: timestamp
- contentUrl: string (Firebase Storage reference to text content)
- sections: array {
    title: string,
    order: number
  }

## Collection: Competitions
- id: string (auto-generated by Firebase)
- name: string
- date: timestamp
- type: string (enum: "invitational", "regional", "state", "national")
- location: string
- description: string
- registrationDeadline: timestamp
- events: string[] (array of available events)
- teamId: string (reference to team participating)
- results: {
    teamRank: number,
    teamOutOf: number,
    eventResults: [
      {
        eventName: string,
        participants: string[] (references to Member ids),
        placement: number,
        outOf: number
      }
    ]
  }

## Collection: Duosmium
- id: string (auto-generated by Firebase)
- url: string (original Duosmium results URL)
- competitionName: string
- date: timestamp
- location: string
- teams: [
    {
      teamName: string (e.g., "Bronx H.S. of Science A NY")
      rank: number
      totalScore: number
      events: [
        {
          eventName: string
          place: number
          participants: string[] (Member IDs who participated)
        }
      ]
    }
  ]
- parsedAt: timestamp 

## Collection: Applications
- id: string (auto-generated by Firebase)
- userId: string (reference to Members)
- competitionId: string (reference to Competitions)
- events: string[] (array of selected events)
- questions: {
    whyCompete: string,
    preparationPlan: string,
    firstTimePrep: string (optional),
    pastInfluence: string (optional)
  }
- status: string (enum: "pending", "approved", "rejected")
- appliedAt: timestamp
- reviewedAt: timestamp (optional)
- reviewNotes: string (optional) 

## Collection: AttendanceEvents
- id: string (auto-generated by Firebase)
- name: string (meeting/event name)
- date: timestamp
- startTime: timestamp
- endTime: timestamp
- description: string (optional)
- location: string
- checkInCode: string (unique code for attendance)
- eventType: string (enum: "meeting", "practice", "competition", "workshop", "other")
- eventFor: string[] (specific events or "all" for everyone)
- createdBy: string (reference to Members - likely leadership)
- createdAt: timestamp
- isActive: boolean

## Collection: Attendance
- id: string (auto-generated by Firebase)
- eventId: string (reference to AttendanceEvents)
- userId: string (reference to Members)
- checkInTime: timestamp
- checkInMethod: string (enum: "code", "admin")
- verifiedBy: string (optional, reference to Members for admin check-ins)
